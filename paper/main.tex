\section{Introduction}
Cerebrospinal fluid (CSF) flow is an important contributor to brain homeostasis
and the regulatory processes of the brain.

In this work, we study intraventricular CSF flow in embryonic zebrafish and adult human brain ventricles.
The geometries are segmented from medical imaging data. First, we simulate CSF flow using a finite
element method. The CSF velocity fields are used to analyze flow topology using particle
trajectories and finite-time Lyapunov exponent fields.

\section{Methods}
\subsection{Finite-time Lyapunov exponent fields}
Consider the dynamical system
\begin{equation}
    \dot{\xx}(t; t_0, \xx_0) = \frac{\mathrm d\xx}{\mathrm d t} = \uu(\xx(t; t_0, \xx_0), t),
    \label{eq:dx_dt_equals_u}
\end{equation}
where $\xx(t)\in$ are the spatial coordinates of the domain $\mathcal{D}\subset\mathbb{R}^d$,
$t$ denotes time, $\uu$ is a velocity, and $\xx_0=\xx(t_0; t_0, \xx_0)$ is the initial
position at the initial time $t_0$. 
\Cref{eq:dx_dt_equals_u} can be integrated to yield the position at a time $T$:
\begin{equation}
    \xx(T) = \xx(t_0) + \int_{t_0}^T \uu(\xx, t)\mathrm d t.
\end{equation}
We define this relation as the mapping
\begin{equation}
    \bphi_{t_0}^t : \mathcal{D} \rightarrow \mathcal{D} : \xx_0 \mapsto \bphi_{t_0}^t(\xx_0) = \xx(t; t_0, \xx_0),
\end{equation}
where $\phi_{t_0}^t$ is the flow map~\cite{shadden2005definition}.
We have thus far emphasized the dependence of 
the dynamical system and the flow map on the parameters $t_0, \xx_0$, but will in the 
following omit this notation for brevity. Using the gradient of the flow map,
we define the (right) Cauchy-Green deformation tensor
\begin{equation}
    \mathbf{C} = \left(\nabla\bphi_{t_0}^T\right)^*\nabla\bphi_{t_0}^T
\end{equation}
where $\mathbf{A}^*$ denotes the transpose of a tensor $\mathbf{A}$. 
In the flow field $\uu(\xx, t)$, the maximum stretching that occurs between
two points $\xx$ and $\yy$ aligns with the eigenvector associated with
the maximum eigenvalue $\lambda_{\mathrm{max}}$ of $\mathbf{C}$~\cite{shadden2005definition}.
The finite-time Lyapunov exponent is defined as
\begin{equation}
    \sigma_{t_0}^T(\xx) = \frac{1}{\vert T\vert}\log{\sqrt{\lambda_{\mathrm{max}}}}
\end{equation}
and provides a measure of the stretching of the flow. Note the absolute value of
the integration time $T$, which is employed because integration forward in time
($T>0$) provides stable manifolds, whereas integration backward in time ($T<0$)
provides unstable manifolds.

\subsection{The governing equations of cerebrospinal fluid flow}
We consider flow of an incompressible, Newtonian fluid of density $\rho$ and dynamic viscosity
$\mu$ in a domain $\mathcal{D}\subset\mathbb{R}^d$ with boundary $\partial\mathcal{D}=\mathcal{S}$.
No body forces are acting on the fluid. The flow is governed by the Navier-Stokes equations,
in which the momentum equation reads
\begin{equation}
\rho\frac{D\uu}{\mathrm Dt} - \nabla\cdot\mathbb{T} = \bm{0},
\end{equation}
where
\begin{equation}
    \rho\frac{D\uu}{\mathrm Dt} = \rho\left(\pdifft{\uu} + (\uu\cdot\nabla)\uu\right)
    \label{eq:material_derivative}
\end{equation}
is the material derivative of the velocity $\uu$ and
the stress tensor $\mathbb{T}$ is defined as
\begin{equation}
    \mathbb{T} = 2\mu\beps(\uu) - p\mathbf{I}.
\end{equation}
Here, the identity tensor in dimension $d$ is $\mathbf{I}$ and
$\beps(\uu)=\frac{1}{2}\left(\nabla\uu + (\nabla\uu)^T\right)$ is the
strain-rate tensor, or symmetric gradient, of $\uu$.
In addition to the momentum equation, mass conservation
is stated with the continuity equation for an
incompressible fluid:
\begin{equation}
    \nabla\cdot\uu = 0.
\end{equation}
We define the Reynolds number
\begin{equation*}
    \mathrm{Re} = \frac{\mathrm{inertial \ forces}}{\mathrm{viscous \ forces}} = \frac{\rho UL}{\mu},
\end{equation*}
where $U$ and $L$ are characteristic velocity and length scales, respectively.
In the case that viscous forces dominate ($\mathrm{Re} \ll 1$), the intertial
terms (\Cref{eq:material_derivative}) can be neglected, yielding the
Stokes equations:
\begin{equation}
    - \nabla\cdot\mathbb{T} = \bm{0}.
\end{equation}

\subsection{The governing equations of Stokes flow}
Due to low Reynolds numbers in CSF flow in zebrafish brain
ventricles, we neglect inertial effects. Thus the left-hand side of~\Cref{eq:navier_stokes}
is dropped yielding the governing equation for Stokes flow:
\begin{equation}
    - \nabla\cdot\bm{\sigma} = \mathbf{0},
    \label{eq:time_dependent_stokes}
\end{equation}
where $\uu$ and $p$ are the fluid velocity and pressure, respectively. The stress tensor $\sigma$ is defined as
\begin{equation*}
    \sigma(\uu, p) = 2\mu\bm{\varepsilon}(\uu) - p\mathbf{I}
\end{equation*}
and
\begin{equation*}
    \bm{\varepsilon}(\uu) = \frac{1}{2}\left(\nabla \uu + (\nabla\uu)^T\right)
\end{equation*}
is the symmetric part of the velocity gradient, or strain-rate tensor. We derive a weak form of~\cref{eq:time_dependent_stokes} by multiplying the equation with a test function $\vv\in\VV$ and integrating over the domain
\begin{align}
    &\intO{\rho\pdifft{\uu}\cdot\vv} - \intO{(\nabla\cdot\bm{\sigma})\cdot\vv} = \mathbf{0} \\
    &\intO{\rho\pdifft{\uu}\cdot\vv} - \intG{\bm{\sigma}\nn\cdot\vv} + \intO{\bm{\sigma} : \nabla\vv} = \mathbf{0} \\
    &\intO{\rho\pdifft{\uu}\cdot\vv} - \intG{\bm{\sigma}\nn\cdot\vv} + \intO{2\mu\varepsilon(\uu) : \varepsilon(\vv)} - \intO{p\nabla\cdot\vv} = \mathbf{0}, \label{eq:weak_form_full}
\end{align}
where line two results from integration by parts of the stress term. 

\subsection{Flow mechanisms and boundary conditions}
We consider three fluid motion-inducing mechanisms, and model them as follows:
\begin{itemize}
    \item Ventricles deformation related to the cardiac and respiratory cycles,
    modeled with an imposed deformation velocity
    \item Secretion of CSF from the choroid plexus,
    modeled with an influx boundary condition on the velocity
    \item Motile cilia beating, modeled as a tangential boundary traction acting on the fluid
     
\end{itemize}

\subsubsection{Brain ventricles deformation}
To model ventricular wall deformations associated with the
cardiac and respiratory cycles, we impose a
deformation velocity $\pdifft{\ww}$ as a boundary
condition when solving the flow equations.
We model the deformation $\ww(\xx, t)$
by imposing a motion on the parts of the geometry,
then solving the equations of linear elasticity, namely
\begin{equation}
    \rho\pdiffdifft{\ww} - \nabla\cdot\bsigma_s = \bm{0} \quad\mathrm{in \ \mathcal{D}},
    \label{eq:strong_linear_elasticity_1}
\end{equation}
where $\bsigma_s = 2\eta\beps(\ww) + \lambda\mathrm{tr}(\beps(\ww))\delta$ is the
solid stress tensor, in which $\eta$ and $\lambda$ are the Lamé parameters of the ventricular wall
and $\delta$ is the Kronecker delta. Applying the divergence operator, \Cref{eq:strong_linear_elasticity_1}
takes the form
\begin{equation}
    \rho\pdiffdifft{\ww} - 2\eta\nabla\cdot\beps(\ww) - \lambda\nabla(\nabla\cdot\ww) = \bm{0}, \quad\mathrm{in \ \mathcal{D}}.
\end{equation}
The boundary conditions are
\begin{align}
    \ww &= \bm{g} \quad\mathrm{on \ \mathcal{S}_1}, \\
    \bsigma_s\nn &= \bm{0} \quad\mathrm{on \ \mathcal{S}_2 = \mathcal{S}\setminus\mathcal{S}_1}.
\end{align}
On $\mathcal{S}_1$ we impose a given deformation $\bm{g}$, whereas
$\mathcal{S}_2$ is considered traction-free.


We assume that the deformation $\mathbf{d}$ only has a $z$ component (feet-head axis).
The displacement is based on measurements~\cite{Kurtcuoglu2007ComputationalSylvius,
Enzmann1992BrainImaging}.

\subsubsection{Choroid plexus secretion}
The secretion is modeled by imposing a normal velocity at the
boundary $\Gamma_{\mathrm{chp}}$ such that the total flux through the choroid plexus boundary is
\begin{equation}
    \int_{\Gamma_{\mathrm{chp}}}(\uu\cdot\nn)\,\mathrm ds = Q.
\end{equation}
Here $Q=504$ ml/day is the daily total CSF production \cite{Czosnyka2004CerebrospinalDynamics}.
From this, we calculate $\uu_\mathrm{in}=Q/\vert\Gamma_{\mathrm{chp}}\vert$.
\begin{equation}
    \uu\cdot\nn = \uu_{\mathrm{in}}\cdot\nn\quad\mathrm{on \ \Gamma_{\mathrm{chp}}},
    \label{eq:choroid_plexus_BC}
\end{equation}
where $\uu_{\mathrm{in}}$ . 

This value is based on the assumption that all CSF that enters the brain ventricles
is produced by the choroid plexus, even though in reality 10--30\% of the CSF production
may come from interstitial fluid transported through the ependymal cell
layer of the brain ventricles walls into the ventricles~\cite{Brodal2016TheSystem}.

\subsubsection{Cilia forces}
Motile cilia are modeled by imposing forces on the fluid with a
boundary condition on the tangential traction
$P_{\nn}(\bsigmahat) = \btau = \tau P_{\nn}(\rr)$.
Here $\bsigmahat=\bsigma\nn$ is the traction vector and
$P_{\nn}(\cdot)$ is the projection of a vector onto the plane which
$\nn$ is normal to. The vector $\btau$ models the cilia motion.
It has been shown that motile cilia beating direction is oriented by
CSF production during neural development~\cite{Guirao2010CouplingCilia}.
A previous study has considered directionality of the cilia forcing aligned
with the wall shear stress of the macroscopic flow simulated in a
production-only setup~\cite{Siyahhan2014FlowVentricles}.
We apply a similar approach, simulating CSF flow with the
boundary condition~\eqref{eq:choroid_plexus_BC} on $\Gchp$,
a zero normal traction condition on the spinal canal and lateral apertures,
and $\uu\cdot\nn=0$ on the rest of the boundary. In the
tangential direction, the CSF is free to slip. The resulting velocity field,
$\uu_{\mathrm{prod}}$, is normalized in length to yield the direction of the cilia:
\begin{equation*}
    \rr = \frac{\uu_{\mathrm{prod}}}{|\uu_{\mathrm{prod}}|}.
\end{equation*}

The parameter $\tau$ determines the magnitude of $\btau$. In previous work on zebrafish modeling,
we determined $\tau$ heuristically by comparing simulated CSF velocity fields with \emph{in vivo}
flow data~\cite{Olstad, Herlyng}.

\subsection{Ventricular geometries and model scenarios}
We consider two geometries: the ventricular geometry of a larval zebrafish at two days
post-fertilization, and the brain ventricles of an adult (healthy) human. In larval zebrafish,
choroid plexus secretion begins around four days post-fertilization. Therefore, we only consider
cilia forces in this flow model. For the human brain, we impose both deformation, choroid plexus
secretion and cilia forces.


\subsection{Numerical methods}
Initially, we solve the Stokes equations and use the solution
$\uu_{\mathrm{Stk}}$ as initial condition for the Navier-Stokes equation;
$\uu_0=\uu(\xx, t=0)=\uu_{\mathrm{Stk}}$. The convective term
is linearized as
\begin{equation*}
    (\uu^{n+1}\cdot\nabla)\uu^{n+1} \approx (\uu^n\cdot\nabla)\uu^{n+1},
\end{equation*}
where $n$ denotes timestep.

We employ a BDM-DG scheme for both the Stokes and the Navier-Stokes equations.

In time, we consider the implicit, second-order discretization scheme BDF2\cite{Volker2016}.

\subsection{Weak formulation}
\subsubsection{Tangential traction}
We impose the tangential stress condition weakly as follows. The stress force vector $\bm{\sigma}\nn$ can be decomposed into a normal and tangential component:
\begin{equation}
    \bm{\sigma}\nn = (\bm{\sigma}\nn\cdot\nn)\nn + P_{\nn}(\bm{\sigma}\nn)
\end{equation}
The boundary integral in~\eqref{eq:weak_form_full} can then be rewritten as
\begin{align}
    \intG{\bm{\sigma}\nn\cdot\vv} &= \intG{(\bm{\sigma}\nn\cdot\nn)\nn\cdot\vv} + \intG{P_{\nn}(\bm{\sigma}\nn)\cdot\vv} \\
    &= \intG{(\bm{\sigma}\nn\cdot\nn)(\vv\cdot\nn)} + \intG{P_{\nn}(\bm{\sigma}\nn)\cdot P_{\nn}(\vv)} \label{eq:bdry_int_decomposed}
\end{align}
Application of boundary conditions in the normal and tangential directions is done by using respectively the first and second terms of~\eqref{eq:bdry_int_decomposed}.

\subsection{Discretization}

We consider an $\mathbf{H}(\mathrm{div})$-conforming discretization of the Stokes equations using
Brezzi-Douglas-Marini (BDM) elements for the velocity and discontinuous Galerkin (DG)
elements for the pressure. The BDM elements have degrees of freedom associated with
the normal component of vectors on the facets of the mesh cells. Thus, setting boundary
conditions strongly for BDM-discretized variables constitutes fixing the normal
component of the discretized variable on a facet. We can therefore impose~\eqref{eq:impermeability_BC}
on the boundary strongly by setting the degrees of freedom to zero. Owing to this strong imposition,
the normal component integral in~\eqref{eq:bdry_int_decomposed} is dropped, because $\vv\cdot\nn=0$ on
the boundary when $\uu\cdot\nn=0$ is imposed. Inserting the tangential stress boundary
condition~\eqref{eq:tangent_stress_BC} into~\eqref{eq:bdry_int_decomposed}, we then have
\begin{equation}
    \intG{\bm{\sigma}\nn\cdot\vv} = \intG{(\bm{\sigma}\nn\cdot\nn)(\vv\cdot\nn)} + \intG{P_{\nn}(\bm{\sigma}\nn)\cdot P_{\nn}(\vv)} = \intG{\bm{\tau}\cdot P_{\nn}(\vv)}.
\end{equation}
The resulting weak form of the momentum equation is
\begin{equation}
    \intO{\rho\pdifft{\uu}\cdot\vv} + \intO{2\mu\varepsilon(\uu) : \varepsilon(\vv)} - \intO{p\nabla\cdot\vv} = \intG{\bm{\tau}\cdot P_{\nn}(\vv)},
    \label{eq:weak_form_cont}
\end{equation}
which is written abstractly as
\begin{equation}
    a(\uu, \vv) + b(p, \vv) = L_s(\vv), \quad\forall\vv\in\VV,
    \label{eq:weak_form_cont_abstract}
\end{equation}
with the definitions
\begin{align}
    a(\uu, \vv) &= \intO{\rho\pdifft{\uu}\cdot\vv} + \intO{2\mu\varepsilon(\uu) : \varepsilon(\vv)}, \\
    b(p, \vv) &= - \intO{p\nabla\cdot\vv}, \\
    L_s(\vv) &= \intG{\bm{\tau}\cdot P_{\nn}(\vv)}.
\end{align}

The continuity equation~\eqref{eq:navier_stokes2} is multiplied by a test function $q\in Q$ and integrated over the domain:
\begin{equation}
    \intO{q\nabla\cdot\uu} = 0, \quad\forall q\in Q.
\end{equation}
We recognize that this equation can be written as $b(\uu, q) = 0$. The resulting weak form of the Stokes equations is then the saddle-point problem
\begin{subequations}
\begin{alignat}{2}
    a(\uu, \vv) + b(p, \vv) &= L_s(\vv), &&\quad\forall\vv\in\VV, \label{eq:mom_weak_abstract}\\
    b(q, \uu) &= 0, &&\quad\forall q\in Q.
\end{alignat}
\label{eq:stokes_weak_form_abstract}%
\end{subequations}%

The discretization considered here is based on the work by Hong~\emph{et al.}~\cite{Hong2016AEquations}. We let the subscript $h$ denote discretized variables. Consider~\eqref{eq:stokes_weak_form_abstract} with discrete variables:
\begin{subequations}
\begin{alignat}{2}
    a(\uu_h, \vv_h) + b(p_h, \vv_h) &= L(\vv_h), &&\quad\forall\vv_h\in\VV_h, \\
    b(q_h, \uu_h) &= 0, &&\quad\forall \: q_h\in Q_h,
\end{alignat}
\label{eq:stokes_weak_form_discrete}%
\end{subequations}%
with the discrete function spaces $\VV_h$ and $Q_h$.
Discretizing the variables in~\eqref{eq:stokes_weak_form_discrete} with $\mathrm{BDM}_k-\mathrm{DG}_{k-1}$ elements, this is not automatically stable discretization. To ensure coercivity and continuity of the bilinear form, we need to add the following stabilization terms to the weak form:
\begin{equation}
    \begin{aligned}
      a_{\mathrm{stab}}(\uu_h, \vv_h) = &-\intE{\avg{2\mu\bm{\varepsilon}(\uu_h)\nn}\cdot\jump{P_{\nn}(\vv_h)}} \\
                                                &-\intE{\avg{2\mu\bm{\varepsilon}(\vv_h)\nn}\cdot\jump{P_{\nn}(\uu_h)}} \\
                                                &+2\mu\frac{\gamma}{h_a}\intE{\jump{P_{\nn}(\uu_h)}\cdot\jump{P_{\nn}(\vv_h)}}, \\
    \end{aligned}
\end{equation}
where $\mathcal{E}$ are all internal facets of the mesh. Furthermore,
$\gamma > 0$ is a penalty parameter which must be chosen large enough and
$h_a$ is the average cell diameter of two neighboring elements.
The jump and average operators $[[\cdot]]$ and $\{\cdot\}$ of a variable $\bm{\phi}$ are defined as
\begin{align}
    \jump{\bm{\phi}} &= \bm{\phi}\vert_{+} - \bm{\phi}\vert_{-}, \\
    \avg{\bm{\phi}}  &= \frac{1}{2}\big((\bm{\phi}\cdot\nn)\vert_{+} - (\bm{\phi}\cdot\nn)\vert_{-} \big),
\end{align}
where $\bm{\phi}\vert_{+}$ is used to denote the value of the variable $\bm{\phi}$ on the side of the edge $\mathcal{E}$ where $\nn$ is pointing out of the cell. Correspondingly, $\bm{\phi}\vert_{-}$ is the variable value on the side of the edge $\mathcal{E}$ where  $\nn$ is pointing inwards.

\section{Verification}
Convergence properties of the discretization scheme were assessed with two
model problems. To assess spatial convergence, we solved a channel flow
problem on increasingly refined meshes. Temporal convergence was assessed
with a Taylor vortex problem~\cite{Karniadakis2005Verification}.
The method converges.

To assess the physiological validity of the flow, we compare some
quantities of interest with previous works~\cite{Kurtcuoglu2007ComputationalSylvius, Causemann2022HumanFramework}:
\begin{itemize}
    \item The pressure gradient in the cerebral aqueduct

    \item The flow rate in the cerebral aqueduct and the outlet to the spinal canal
\end{itemize}
